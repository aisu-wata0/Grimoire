66  * @param ppElement pointer to array (pointer)

68  * @param boundary : Power of two, else the behavior is undefined
69  * @return The pointer you should free, don't lose it

71 template<typename elem>
72 void* al_allloc(elem** ppElement, size_t size, size_t boundary){
73  *ppElement = (elem*)malloc((size)*sizeof(elem) + boundary-1);
74  if(*ppElement == NULL){
75  cerr <<"failed to malloc "<< (size)*sizeof(elem)/1024 <<" KiB"<< endl;
76  exit(0);
77  }
78  void* pMem = *ppElement;
79  *ppElement = (elem*)(((uintptr_t)pMem + (boundary-1)) & (uintptr_t)( ~ (boundary-1)));
80  return pMem;
81 }

83  * @brief Calculated padded size
84  * to align the end to a cache line and avoid cache trashing
85  */
86 size_t calcPadSize(size_t size){
87  // add to make it multiple of L1_LINE_DN (padding)
88  size = roundUpMultiple(size, L1_LINE_DN);
89  // make sure size is not a power of two, avoid cache trashing
90  if(size > L1LINE_N-1 && isPowerOfTwo(size))
91  size += L1_LINE_DN;

94 /**
95  * @brief Vectorized array, use this to use Vector Extensions easily \n
96  * Uses dynamic allocated aligned memory. The start of the array is 64 bytes aligned \n
97  * so the vectorization can be used in a loop from 0 to size()/vecN()
98  * (== vecInd(size()) == sizeVec()) \n
99  * the remaining non vectorized elements are looped
100  * from sizeVec()*vecN() (== remStart() == remInd(sizeVec())).
101  * If you must loop from a index that is not multiple of vecN()
102  * you have to loop through the first indexes until it reaches a multiple of 4. \n
103  * Example for a generic loop from start to end: \n
104  * ```cpp
105  * fstEnd = roundUpMultiple(start, vecN());
106  * for(i = start; i < fstEnd; ++i) // fsrt loop // if start%vecN == 0, no loop
107  * something(V[i]);
108  * for(iv = V.vecInd(i); iv < V.vecInd(end); ++iv) // vec loop
109  * something(V.atv(vi));
110  * for(i = V.remInd(vi); i < end; ++i) // remaining loop // if end%vecN == 0, no loop
